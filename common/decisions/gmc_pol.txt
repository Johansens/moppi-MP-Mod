seek_british_aid = {
	is_shown = {
		exists = c:POL
		c:POL = ROOT
		is_player = yes
		hidden_trigger = {
			Not = {has_variable = british_aid_taken}
		}
	}

	possible = {
		hidden_trigger = {
			c:POL ?= ROOT
		}

		c:GBR ?= {
				has_variable = poland_aid
			}

		Not = {	
			c:POL ?= {
				has_variable = aid_talks
			}
		}
	
}
	when_taken = {

		     c:GBR ?= {
				trigger_event = { id = gmc_polish_events.3 }
			}

		set_variable = british_aid_taken
		set_variable = aid_talks
	}

	ai_chance = {
		base = -100		
	}
}
