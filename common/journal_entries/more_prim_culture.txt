je_arg_more_prim_culture = { ## Here, a new primary culture is added to each of 5 countries
	icon = "gfx/interface/icons/event_icons/event_map.dds"

	group = je_group_internal_affairs

	is_shown_when_inactive = {
    c:ARG ?= ROOT
	}

  possible = {
    c:ARG ?= ROOT
  }

  complete = {
    game_date >= 1836.2.2
    }

  on_complete = {
    add_primary_culture = cu:patagonian
  }
}
