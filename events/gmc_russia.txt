	}

			save_scope_as = nicholas_scope
		}

		s:STATE_LESSER_POLAND = {
            region_state:RUS = {
                set_variable = {
                name = great_polish_revolt
                }
            }
        }
		s:STATE_MAZOVIA = {
            region_state:RUS = {
                set_variable = {
                name = great_polish_revolt
                }
            }
        }
		s:STATE_VILNIUS = {
            region_state:RUS = {
                set_variable = {
                name = great_polish_revolt
                }
            }
        }
		s:STATE_KAUNAS = {
            region_state:RUS = {
                set_variable = {
                name = great_polish_revolt
                }
            }
        }
		s:STATE_BREST = {
            region_state:RUS = {
                set_variable = {
                name = great_polish_revolt
                }
            }
        }
		s:STATE_VOLHYNIA = {
            region_state:RUS = {
                set_variable = {
                name = great_polish_revolt
                }
            }
        }
		s:STATE_GREATER_POLAND = {
            region_state:RUS = {
                set_variable = {
                name = great_polish_revolt
                }
            }
        }
		s:STATE_RIGA = {
            region_state:RUS = {
                set_variable = {
                name = great_riga_revolt
                }
            }
        }

		add_modifier = {name = modifier_polish_uprising_russia years = 1}
	}

	option = { # Great rebel uprising
		name = russia_polish_unrest.7.a
		type = country_event
		placement = root

		default_option = yes
		ai_chance = {
			base = 8 # this is the default pick for the AI
		}

		ruler = {
			kill_character = yes
		}

		create_country = {
			tag = POL
			origin = root
			state = s:STATE_GREATER_POLAND.region_state:RUS

			on_created = {
                c:POL = {
                    activate_law = law_type:law_presidential_republic
					activate_law = law_type:law_national_militia
					activate_law = law_type:law_appointed_bureaucrats
					activate_law = law_type:law_freedom_of_conscience
					activate_law = law_type:law_racial_segregation
					activate_law = law_type:law_autocracy
					activate_law = law_type:law_homesteading
					activate_law = law_type:law_land_based_taxation
                }
				s:STATE_MINSK = {
            		add_claim = c:POL
        		}
        		s:STATE_VITEBSK = {
            		add_claim = c:POL
        		}
        		s:STATE_MOGILEV = {
            		add_claim = c:POL
        		}
        		s:STATE_KIEV = {
           			add_claim = c:POL
        		}

				s:STATE_GREATER_POLAND = {
           			add_claim = c:RUS
        		}
				s:STATE_VOLHYNIA = {
           			add_claim = c:RUS
        		}
				s:STATE_BREST = {
           			add_claim = c:RUS
        		}
				s:STATE_MAZOVIA = {
           			add_claim = c:RUS
        		}
				s:STATE_LESSER_POLAND = {
           			add_claim = c:RUS
        		}
				s:STATE_VILNIUS = {
           			add_claim = c:RUS
        		}
				s:STATE_KAUNAS = {
           			add_claim = c:RUS
        		}
				set_variable = {
             		name = polish_great_rebellion
        		}
			}	
    	}

		hidden_effect = {
			c:UBD ?= { 
				make_independent = yes
				add_primary_culture = cu:lithuanian
				add_primary_culture = cu:latvian
				add_primary_culture = cu:estonian
				remove_primary_culture = cu:north_german
				remove_primary_culture = cu:russian

				activate_law = law_type:law_racial_segregation
				activate_law = law_type:law_homesteading
				activate_law = law_type:law_autocracy
				activate_law = law_type:law_presidential_republic
				activate_law = law_type:law_national_militia
			}

			c:POL ?= {
				create_diplomatic_pact = {
				country = c:UBD
				type = puppet
				}
			}
		}

		every_scope_state = {
            limit = {
                is_capital = no
                has_variable = great_riga_revolt
            }     
            set_state_owner = c:UBD
            set_state_type = incorporated
        }
	
		every_scope_state = {
            limit = {
                is_capital = no
                has_variable = great_polish_revolt
                }     
            set_state_owner = c:POL
            set_state_type = incorporated
        }
		c:RUS = {
				create_diplomatic_play = {
					name = great_polish_uprising
					type = dp_return_state
					target_state = s:STATE_GREATER_POLAND.region_state:POL
					target_country = c:POL

					add_war_goal = {
						holder = c:RUS
						type = return_state
						target_state = s:STATE_VOLHYNIA.region_state:POL
					}
					add_war_goal = {
						holder = c:RUS
						type = return_state
						target_state = s:STATE_BREST.region_state:POL
					}
					add_war_goal = {
						holder = c:RUS
						type = return_state
						target_state = s:STATE_MAZOVIA.region_state:POL
					}
					add_war_goal = {
						holder = c:RUS
						type = return_state
						target_state = s:STATE_LESSER_POLAND.region_state:POL
					}
					add_war_goal = {
						holder = c:RUS
						type = return_state
						target_state = s:STATE_VILNIUS.region_state:POL
					}
					add_war_goal = {
						holder = c:RUS
						type = return_state
						target_state = s:STATE_KAUNAS.region_state:POL
					}				 

					add_war_goal = {
						holder = c:POL
						type = return_state
						target_state = s:STATE_VITEBSK.region_state:RUS
					}
					add_war_goal = {
						holder = c:POL
						type = return_state
						target_state = s:STATE_MOGILEV.region_state:RUS
					}
					add_war_goal = {
						holder = c:POL
						type = return_state
						target_state = s:STATE_KIEV.region_state:RUS
					}
					add_war_goal = {
						holder = c:POL
						type = return_state
						target_state = s:STATE_MINSK.region_state:RUS
					}
				}
			}
		trigger_event = { id = russia_polish_unrest.12 popup = yes }
	}
}
