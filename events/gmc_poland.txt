	immediate = {
				
	}

	duration = 3

	option = { # help, but at a high price
		name = gmc_polish_events.3.a

        default_option = yes

			base = 6
		}

        c:POL ?= {
            trigger_event = { id = gmc_polish_events.4 popup = yes}
        }
		custom_tooltip = british_help_high_price_tt
		show_as_tooltip = {
			c:POL ?= {
        		add_technology_researched = lathe
        		add_technology_researched = atmospheric_engine
        		add_technology_researched = railways
    	    	add_technology_researched = mechanical_tools
    	    	add_technology_researched = percussion_cap
    	    	add_technology_researched = general_staff
    	    	add_technology_researched = banking
    	    	add_technology_researched = stock_exchange
    	    	add_technology_researched = mass_communication
    	    	add_technology_researched = empiricism
			}
			c:GBR ?= {
				create_diplomatic_pact = {
					country = c:POL
					type = protectorate
				}
			}
		}
	}

    option = { # help, at a small price, but not quite as strong
		name = gmc_polish_events.3.b

        ai_chance = {
			base = 3
		}

        c:POL ?= {
            trigger_event = { id = gmc_polish_events.7 popup = yes}
			hidden_effect = {
				remove_variable = aid_talks
			}
        }

		custom_tooltip = british_help_small_price_tt
		show_as_tooltip = {
			c:POL ?= {
        		add_technology_researched = lathe
        		add_technology_researched = banking
        		add_technology_researched = stock_exchange
        		add_technology_researched = mass_communication
        		add_technology_researched = empiricism
			}

			c:GBR ?= {
			add_modifier = {name = modifier_gbr_polish_payments years = 5}
			}
			c:POL ?= {
			add_modifier = {name = modifier_pol_polish_payments years = 5}
			}
			}
	}
    option = { # request for help rejected
		name = gmc_polish_events.3.c

        ai_chance = {
			base = 1
		}

        c:POL ?= {
            trigger_event = { id = gmc_polish_events.8 popup = yes}

			}
        }

			c:POL ?= {
        		change_relations = {
				country = c:GBR
				value = -20
				}
			}
		}
